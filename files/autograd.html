import math
from graphviz import Digraph

class Value:
    def __init__(self,data,_children=(),_op="",_name="",requires_grad = True):
        self.data = data
        # 子节点
        self._prev = set(_children)
        # 运算符
        self._op = _op
        # 偏导
        self.grad = 0.0
        self._name = _name
        # 常数则不需要偏导
        self.requires_grad = requires_grad
        self._backward = lambda:None

    def __repr__(self):
        """
        数据展示
        """
        return f"Value(data={self.data})"

    """
    运算符定义backward函数本质就是再已知dL/dout的情况下，求dL/dself & dL / dother，基于out 和 self & other的关系来确定
    """
    
    def __add__(self,other):
        # 需要考虑常数，如果是常数，那么就不需要偏导
        other = other if isinstance(other,Value) else Value(other,requires_grad = False)
        # 运算结果
        out = Value(self.data + other.data,(self,other),'+')
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                self.grad += out.grad
            if other.requires_grad:
                other.grad += out.grad
        out._backward = backward
        return out

    def __mul__(self,other):
        other = other if isinstance(other,Value) else Value(other,requires_grad = False)
        out = Value(self.data * other.data,(self,other),'*')
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                self.grad += out.grad * other.data
            if other.requires_grad:
                other.grad += out.grad * self.data
        out._backward = backward
        return out

    def __sub__(self,other):
        other = other if isinstance(other,Value) else Value(other,requires_grad = False)
        out = Value(self.data - other.data,(self,other),'-')
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                self.grad += out.grad
            if other.requires_grad:
                other.grad -= out.grad
        out._backward = backward
        return out

    def __truediv__(self,other):
        other = other if isinstance(other,Value) else Value(other,requires_grad = False)
        out = Value(self.data / other.data,(self,other),'/')
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                self.grad += out.grad * (1 / other.data)
            if other.requires_grad:
                other.grad += out.grad * (-self.data /(other.data ** 2))
        out._backward = backward
        return out

    def __radd__(self,other):
        return self + other

    def __rmul__(self,other):
        return self * other

    def __rsub__(self, other): # other - self
        return Value(other) - self

    def __rtruediv__(self, other): # other / self
        return Value(other) / self

    def __neg__(self):
        return self * -1

    def tanh(self):
        x = self.data
        t = math.tanh(x)
        out = Value(t,(self,),"tanh")
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                self.grad += out.grad * (1 - t ** 2)
        out._backward = backward
        return out

    def exp(self):
        x = self.data
        out = Value(math.exp(x),(self,),"exp")
        def backward():
            # 只有需要求偏导的才传导
            if self.requires_grad:
                # d/dx (e^x) = e^x = out.data
                self.grad += out.grad * out.data
        out._backward = backward
        return out

    def backward(self):
        #去重，避免访问重复的节点，链规则跑一次就可以
        visited = set()
        #深度优先，记录路上的点，最后要按照传递顺序回填偏导
        topo_list = []
        def topo(v):
            if v not in visited:
                #加入集合，避免重复
                visited.add(v)
                #dfs算法
                for child in v._prev:
                    topo(child)
                #记录路上的点，不过是按照叶子最先，根最后的顺序，我们后面要按照从根开始回填
                topo_list.append(v)
        topo(self)
        #防止多次 backward 导致梯度累加
        for node in topo_list:
            node.grad = 0.0
        # dl / dl = 1.0
        self.grad = 1.0
        for node in reversed(topo_list):
            node._backward()


def draw_graph(root, filename="computational_graph"):
    """
    绘制以 root 为根节点的计算图。
    返回 Digraph 对象，在 Jupyter 中可直接显示。
    """
    dot = Digraph(format='svg', graph_attr={'rankdir': 'LR'})
    #集合避免生成重复的点
    visited = set()

    def get_id(v):
        """
        生成node所需要的id来区分node
        """
        return f"v_{id(v)}"

    def traverse(root):
        """
        类似backward的深度优先算法
        """
        if root in visited:
            return
        visited.add(root)
        # 处理node
        label = root._name if root._name else f"data: {root.data:.4f}"
        if root.grad != 0.0:
            label += f"|grad: {root.grad:.4f}"
        dot.node(get_id(root),label = label,shape = 'record')
        if root._op:
            op_node_id = f"op_{id(root)}_{root._op}"
            dot.node(op_node_id,label = root._op,shape = 'circle')
            # 连接运算符节点到当前 Value 节点
            dot.edge(op_node_id, get_id(root))

            for child in root._prev:
                traverse(child)
                #连接子节点到运算符节点
                dot.edge(get_id(child), op_node_id)

    traverse(root)
    return dot
